```javascript

const date = new Date()
const year = date.getFullYear()
const month = date.getMonth()

// Get first day of the month 1/X
const firstDayOfMonth = new Date(year, month, 1)

// Get the day (1: Monday, 2: Tuesday, 3: Wednesday...)
const dayOfWeek = firstDayOfMonth.getDay()

// Get days to substract
// eg: Friday (4) - 3 = Monday (1)
const substractDays = dayOfWeek ? (dayOfWeek - 1) : 6

new Date(year, month, 1 - substractDays)
```
